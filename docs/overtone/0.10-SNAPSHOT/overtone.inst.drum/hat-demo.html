      (definst hat-demo
  [amp    {:default 0.3 :min 0.001 :max 1 :step 0.01}
   t      {:default 0.3 :min 0.1 :max 1.0 :step 0.01}]
  (let [low (lpf (white-noise) (mouse-x 3000 12000))
        hi (hpf low (mouse-y 1000 8000))
        env (line 1 0 t :action FREE)]
    (* amp env hi)))
